    const apiKey = ""; 
    let base64FileData = "";
    let fileMimeType = "";
    let originalFileName = "";
    const MAX_QUESTIONS_PER_CHUNK = 60; // رقم آمن جداً للتقسيم

    // نظام الـ Logging الاحترافي
    function logToTerminal(message, type = 'info') {
        terminal.style.display = 'block';
        const p = document.createElement('p');
        const time = new Date().toLocaleTimeString('en-US', { hour12: false });
        p.className = `log-${type}`;
        p.innerHTML = `<span style="color:#7f8c8d;">[${time}]</span> ${message}`;
        terminal.appendChild(p);
        terminal.scrollTop = terminal.scrollHeight; // التمرير التلقائي للأسفل
    }

    function getPreciseMimeType(file) {
        if (file.type) return file.type;
        const ext = file.name.split('.').pop().toLowerCase();
        if (ext === 'pdf') return 'application/pdf';
        if (ext === 'txt') return 'text/plain';
        if (ext === 'docx') return 'application/vnd.openxmlformats-officedocument.wordprocessingml.document';
        if (ext === 'pptx') return 'application/vnd.openxmlformats-officedocument.presentationml.presentation';
        return 'application/octet-stream';
    }

    fileInput.addEventListener('change', (e) => {
        const file = e.target.files[0];
        terminal.innerHTML = '';
        terminal.style.display = 'none';
        
        if (file) {
            originalFileName = file.name;
            fileMimeType = getPreciseMimeType(file);
            fileNameDisplay.innerText = `الملف المختار: ${file.name}`;
            generateBtn.style.display = 'block';
            
            const reader = new FileReader();
            reader.onload = (event) => {
                base64FileData = event.target.result.split(',')[1];
                logToTerminal(`File loaded into memory: ${file.name} (${Math.round(file.size/1024)} KB)`, 'system');
            };
            reader.readAsDataURL(file);
        }
    });

    generateBtn.addEventListener('click', async () => {
        if (!base64FileData) {
            logToTerminal("Error: File data missing.", 'error');
            return;
        }

        generateBtn.disabled = true;
        generateBtn.style.opacity = '0.7';
        loader.style.display = 'flex';
        terminal.innerHTML = '';

        try {
            // المرحلة الأولى: الوكيل المُحلل (Analyzer Agent)
            logToTerminal("Phase 1: Dispatching Analyzer Agent to map the document...", 'info');
            const plan = await analyzeDocumentPlan(base64FileData, fileMimeType);
            
            if (!plan.hasAnswers) {
                throw new Error("Analyzer Report: No clear questions or answer key found. Operation aborted.");
            }

            logToTerminal(`Analyzer Report: Found ${plan.totalQuestions} questions.`, 'success');
            logToTerminal(`Analyzer Report: Splitting task into ${plan.chunks.length} chunks to prevent Token Limit Overflow.`, 'warning');

            // المرحلة الثانية: المعالجة المتوازية (Parallel Extractor Agents)
            logToTerminal("Phase 2: Dispatching Extractor Agents in parallel...", 'info');
            
            // إنشاء وعود (Promises) لكل جزء
            const extractionPromises = plan.chunks.map((chunk, index) => {
                logToTerminal(`Agent #${index+1} assigned to range: Q${chunk.start} to Q${chunk.end}`, 'system');
                return extractChunk(base64FileData, fileMimeType, chunk.start, chunk.end, index + 1);
            });

            // انتظار انتهاء جميع الوكلاء بالتوازي
            const chunksResults = await Promise.all(extractionPromises);
            
            // دمج كل المصفوفات في مصفوفة واحدة
            let allExtractedQuestions = [];
            chunksResults.forEach(chunkArray => {
                allExtractedQuestions = allExtractedQuestions.concat(chunkArray);
            });

            logToTerminal(`Phase 3: Aggregation complete. Total validated questions: ${allExtractedQuestions.length}`, 'success');

            if (allExtractedQuestions.length > 0) {
                logToTerminal("Generating final interactive HTML file...", 'info');
                createAndDownloadHTMLFile(allExtractedQuestions, originalFileName);
                logToTerminal("Mission Accomplished! File downloaded.", 'success');
            } else {
                throw new Error("Critical Error: Aggregation resulted in 0 questions.");
            }

        } catch (error) {
            logToTerminal(`FATAL ERROR: ${error.message}`, 'error');
        } finally {
            generateBtn.disabled = false;
            generateBtn.style.opacity = '1';
            loader.style.display = 'none';
        }
    });

    // الوكيل 1: المحلل (Analyzer)

    async function analyzeDocumentPlan(base64Data, mimeType) {
        const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
        
        const prompt = `
        أنت وكيل مُحلل (Analyzer Agent). هدفك قراءة المستند المرفق بالكامل وتحليله.
        1. هل يحتوي المستند على أسئلة اختيار من متعدد بإجاباتها؟
        2. ما هو إجمالي عدد الأسئلة المتوفرة تقريباً؟ (عدها بدقة).
        3. قم بوضع خطة لتقسيم هذه الأسئلة إلى أجزاء (Chunks) بحيث لا يتعدى كل جزء ${MAX_QUESTIONS_PER_CHUNK} سؤالاً.
        
        مثال للخطة (إذا كان هناك 80 سؤال):
        [{"start": 1, "end": 60}, {"start": 61, "end": 80}]
        
        استجب بملف JSON مطابق للـ Schema المطلوبة فقط.
        `;

        const payload = {
            contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: mimeType, data: base64Data } }] }],
            generationConfig: {
                responseMimeType: "application/json",
                responseSchema: {
                    type: "OBJECT",
                    properties: {
                        hasAnswers: { type: "BOOLEAN" },
                        totalQuestions: { type: "INTEGER" },
                        chunks: {
                            type: "ARRAY",
                            items: {
                                type: "OBJECT",
                                properties: {
                                    start: { type: "INTEGER" },
                                    end: { type: "INTEGER" }
                                },
                                required: ["start", "end"]
                            }
                        }
                    },
                    required: ["hasAnswers", "totalQuestions", "chunks"]
                }
            }
        };

        return await makeApiCallWithRetry(url, payload, "Analyzer Agent");
    }

    // -------------------------------------------------------------
    // الوكيل 2: المستخرج (Extractor)
    // -------------------------------------------------------------
    async function extractChunk(base64Data, mimeType, start, end, agentId) {
        const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
        
        const prompt = `
        أنت وكيل استخراج دقيق (Extractor Agent #${agentId}).
        مهمتك: استخراج الأسئلة من رقم ${start} إلى رقم ${end} فقط من المستند المرفق.
        
        شروط صارمة:
        1. استخرج الأسئلة الواقعة في هذا النطاق فقط (${start} إلى ${end}). تجاهل أي سؤال آخر.
        2. الإجابات الصحيحة في هذا الملف محددة بعلامة (•) بجانب الحرف. أو استخدم مفتاح الإجابات.
        3. عدد الخيارات مرن (قد يكون 4 أو 5). استخرج كل الخيارات لكل سؤال.
        4. تجاهل أي أسئلة مقالية شاذة.
        
        استجب بملف JSON يحتوي على الأسئلة المستخرجة:
        "q": نص السؤال، "a": الخيارات (مصفوفة)، "c": رقم الفهرس للإجابة الصحيحة.
        `;

        const payload = {
            contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: mimeType, data: base64Data } }] }],
            generationConfig: {
                maxOutputTokens: 8192, 
                responseMimeType: "application/json",
                responseSchema: {
                    type: "ARRAY",
                    items: {
                        type: "OBJECT",
                        properties: {
                            q: { type: "STRING" },
                            a: { type: "ARRAY", items: { type: "STRING" } },
                            c: { type: "INTEGER" }
                        },
                        required: ["q", "a", "c"]
                    }
                }
            }
        };

        const result = await makeApiCallWithRetry(url, payload, `Extractor Agent #${agentId} (Q${start}-${end})`);
        logToTerminal(`Agent #${agentId} successfully extracted ${result.length} questions.`, 'success');
        return result;
    }

    // -------------------------------------------------------------
    // محرك الاتصال المرن (معالجة الأخطاء والتكرار)
    // -------------------------------------------------------------
    async function makeApiCallWithRetry(url, payload, agentName) {
        // فترات انتظار تصاعدية لمعالجة ضغط السيرفر (Rate Limits)
        const delays = [1000, 2000, 4000, 8000, 16000]; 
        
        for (let i = 0; i <= delays.length; i++) {
            try {
                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                if (!response.ok) {
                    if (response.status === 403) throw new Error("403 Forbidden: Invalid API Key.");
                    if (response.status === 429) {
                        logToTerminal(`[${agentName}] Rate Limit Hit (429). Backing off...`, 'warning');
                        throw new Error("Rate Limit"); // لكي يتم التقاطه وإعادة المحاولة
                    }
                    throw new Error(`Server Error: ${response.status}`);
                }
                
                const data = await response.json();
                const textResult = data.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (!textResult) throw new Error("Empty response from AI.");
                
                try {
                    return JSON.parse(textResult);
                } catch (parseError) {
                    logToTerminal(`[${agentName}] JSON Parse Error. Attempting Auto-Repair...`, 'warning');
                    // نظام الترميم التلقائي في حال الانقطاع المفاجئ
                    const isArray = payload.generationConfig.responseSchema.type === "ARRAY";
                    const lastValidIndex = isArray ? textResult.lastIndexOf('}') : textResult.lastIndexOf('}');
                    
                    if (lastValidIndex !== -1) {
                        const repairedText = isArray ? 
                            textResult.substring(0, lastValidIndex + 1) + ']' : 
                            textResult.substring(0, lastValidIndex + 1) + ']}';
                        return JSON.parse(repairedText);
                    } else {
                        throw parseError;
                    }
                }
                
            } catch (error) {
                if (error.message.includes("403") || i === delays.length) {
                    logToTerminal(`[${agentName}] Failed after ${i} retries: ${error.message}`, 'error');
                    throw error; 
                }
                logToTerminal(`[${agentName}] Retrying in ${delays[i]/1000}s... (${error.message})`, 'warning');
                await new Promise(res => setTimeout(res, delays[i]));
            }
        }
    }

<script>
const allQuestions = ${questionsJSONString};
let currentQuestions = [...allQuestions];
let userAnswers = [];
let currentIdx = 0;

function initQuiz() {
    const container = document.getElementById('questions-container');
    const nav = document.getElementById('nav-panel');
    container.innerHTML = '';
    nav.innerHTML = '';
    userAnswers = new Array(currentQuestions.length).fill(null);
    currentIdx = 0;

    currentQuestions.forEach((q, i) => {
        const btn = document.createElement('button');
        btn.className = 'nav-btn';
        btn.id = \`nav-\${i}\`;
        btn.innerText = i + 1;
        btn.onclick = () => jump(i);
        nav.appendChild(btn);

        const box = document.createElement('div');
        box.className = \`question-box \${i===0?'active':''}\`;
        box.id = \`q-box-\${i}\`;
        
        let opts = q.a.map((opt, oi) => \`
            <button class="option-btn" id="q-\${i}-o-\${oi}" onclick="select(\${i}, \${oi})">\${opt}</button>
        \`).join('');

        box.innerHTML = \`
            <h3>Q \${i+1}:</h3>
            <p style="font-size: 18px;">\${q.q}</p>
            <div class="options">\${opts}</div>
            <div class="feedback" id="feed-\${i}"></div>
        \`;
        container.appendChild(box);
    });
    updateUI();
}

function select(qIdx, oIdx) {
    if (userAnswers[qIdx] !== null) return;
    userAnswers[qIdx] = oIdx;
    const correct = currentQuestions[qIdx].c;
    const btn = document.getElementById(\`q-\${qIdx}-o-\${oIdx}\`);
    const navBtn = document.getElementById(\`nav-\${qIdx}\`);
    const feed = document.getElementById(\`feed-\${qIdx}\`);
    
    feed.style.display = 'block';
    if (oIdx === correct) {
        btn.classList.add('selected-correct');
        navBtn.classList.add('correct');
        feed.innerHTML = "Correct! ✨";
        feed.style.color = "var(--success)";
    } else {
        btn.classList.add('selected-wrong');
        document.getElementById(\`q-\${qIdx}-o-\${correct}\`).classList.add('show-correct');
        navBtn.classList.add('wrong');
        feed.innerHTML = "Incorrect.";
        feed.style.color = "var(--danger)";
    }
    if (userAnswers.every(a => a !== null)) document.getElementById('finish-btn').style.display = 'block';
}

function move(step) {
    let target = currentIdx + step;
    if (target >= 0 && target < currentQuestions.length) jump(target);
}

function jump(idx) {
    document.querySelectorAll('.question-box').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
    document.getElementById(\`q-box-\${idx}\`).classList.add('active');
    document.getElementById(\`nav-\${idx}\`).classList.add('active');
    currentIdx = idx;
    updateUI();
}

function updateUI() {
    document.getElementById('prev-btn').disabled = currentIdx === 0;
    document.getElementById('next-btn').disabled = currentIdx === currentQuestions.length - 1;
}

function showResults() {
